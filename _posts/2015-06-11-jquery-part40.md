#### resolve的处理

* 当最后一个Deferred对象被resolve时，会触发$.when().done(callback)中done中回调的执行

```javascript
var fn1 = function () {
    var dtd = $.Deferred();
    setTimeout(function () {
        console.log('fn1');
        dtd.resolve();
    }, 1000);
    return dtd.promise();
};

var fn2 = function () {
    var dtd = $.Deferred();
    setTimeout(function () {
        console.log('fn2');
        dtd.resolve();
    }, 2000);
    return dtd.promise();
};

var callback = function() {
    console.log('callback');
};

$.when(fn1(), fn2()).done(callback);

// output
// fn1
// fn2
// callback
```

*实现代码*

```javascript
when: function(subordinate) {
    var i = 0,
        resolveValues = slice.call(arguments),
        length = resolveValues.length,
        deferred = jQuery.Deferred(),

        // 每次当外部Deferred对象被resovle时，会调用内部的匿名函数
        // 每调用一次remaining减一，当到最后一个外部Deferred对象resovle时，会触发deferred.resolve()
        updateFunc = function() {
            return function() {
                // remaining为0时，deferred.resolve()
                if (!(--remaining)) {
                    deferred.resolve();
                }
            };
        };

    if (length > 1) {
        for (; i < length; i++) {
            if (resolveValues[i] && jQuery.isFunction(resolveValues[i].promise)) {
                resolveValues[i].promise()
                    // updateFunc()返回一个匿名函数的引用
                    .done(updateFunc())
                    .fail(deferred.reject);
            } else {
                --remaining;
            }
        }
    }

    return deferred.promise();
}
```
*这里不引用jQuery库，模拟整个流程*

```javascript
var fn1 = function () {
    var dtd = $.Deferred();
    setTimeout(function () {
        console.log('fn1');
        dtd.resolve();
    }, 1000);
    return dtd.promise();
};

var fn2 = function () {
    var dtd = $.Deferred();
    setTimeout(function () {
        console.log('fn2');
        dtd.resolve();
    }, 2000);
    return dtd.promise();
};

var callback = function() {
    console.log('callback');
};

// jQuery内部实现

// 1. 在内部创建了一个新的Deferred对象deferred
var deferred = $.Deferred();
// 2. remaining记录Deferred对象个数
var remaining = 2;

var updateFunc = function() {
    return function() {
        if (!(--remaining)) {
            deferred.resolve();
        }
    };
};

// 3. 遍历Deferred对象，把updateFunc中匿名函数的引用作为done的参数
fn1().promise().done(updateFunc());
fn2().promise().done(updateFunc());

// 4. 等同于$.when(fn1(), fn2()).done(callback)
deferred.promise().done(callback);
```

#### 增加参数传递

```javascript
var fn1 = function () {
    var dtd = $.Deferred();
    setTimeout(function () {
        console.log('fn1');
        dtd.resolve('value1');
    }, 1000);
    return dtd.promise();
};

var fn2 = function () {
    var dtd = $.Deferred();
    setTimeout(function () {
        console.log('fn2');
        dtd.resolve('value2');
    }, 2000);
    return dtd.promise();
};

var callback = function() {
    console.log(arguments);
};

$.when(fn1(), fn2()).done(callback);

// output
// fn1
// fn2
// ['value1', 'value2']
```

核心源码：

```javascript
var updateFunc = function(i, values) {
    return function(value) {
        values[i] = arguments.length > 1 ? slice.call(arguments) : value;
        if (!(--remaining)) {
            // values是每个resolve参数组成的数组
            deferred.resolveWith(null, values);
        }
    };
}
```


fire和fireWith一点小差异，之前没有注意到

```javascript
var callback = $.Callbacks();

var fn = function() {
    console.log(arguments);
};

callback.add(fn);

callback.fireWith(this, ['value1', 'value2']);  //['value1', 'value2']

callback.fire(['value1', 'value2']);  //[['value1', 'value2']]
```

部分源码：

```javascript
fireWith: function(context, args) {
    if (list) {
        args = args || [];
        args = [context, args.slice ? args.slice() : args];
        fire(args);
    }
    return this;
},

fire: function() {
    self.fireWith(this, arguments);
    return this;
}
```

* 把数组直接传给fireWith，args[1]是一个数组['value1', 'value2']
* 传给fire，在间接调用fireWith，args[1]为数组的第一个元素'value1'
